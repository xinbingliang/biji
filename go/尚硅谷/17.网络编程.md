# 网络编程

## 流程

1. 监听端口8888
2. 接收客户端Tcp链接，建立客户端和服务端的链接
3. 创建goroutine处理请求

`````
package main

import (
	"fmt"
	"net"
)

func main() {
	fmt.Println("服务器开始监听了。。。。")
	listen, err := net.Listen("tcp", "0.0.0.0:8888")
	if err != nil {
		fmt.Println("监听失败", err)
		return
	}
	defer listen.Close() //确保最后退出

	// 循环等待客户端的链接
	for  {
		//等待链接
		fmt.Println("等待链连")
		conn, err := listen.Accept()
		if err != nil{
			fmt.Println("链接失败")
		} else {
			fmt.Println(conn)
			//启动协程
		}
	}

	fmt.Println("listen = %v", listen)
}

`````
