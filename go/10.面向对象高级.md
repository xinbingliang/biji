## 继承

* 同类对象代码复用问题

  `````go
  package main

  import "fmt"

  type student struct {
  	Name string
  	Age int
  	Score int
  }

  func (stu *student) ShowInfo () {
  	fmt.Printf("学生名=%v，年龄=%v，成绩=%v\n", stu.Name, stu.Age, stu.Score)
  }

  type Pupil struct {
  	student //嵌入student匿名结构体
  }

  type Graduate struct {
  	student
  }

  func (p *Pupil) testing()  {
  	fmt.Println("小学生考试中......")
  	p.Score = 100
  }

  func (p *Graduate) testing()  {
  	fmt.Println("大学生考试中......")
  	p.Score = 97
  }

  func main() {
  	var pupil = &Pupil{}
  	pupil.student.Name = "崔"
  	pupil.student.Age = 8
  	pupil.testing()
  	pupil.ShowInfo()

  	var graduate = &Graduate{}
  	graduate.student.Name = "学"
  	graduate.student.Age = 8
  	graduate.testing()
  	graduate.ShowInfo()
  }
  `````

## 接口

````go
package main

type Usb interface {
	Start()
	Stop()
}

func main() {
}
````

降低耦合性，提高通用性

* 定义一组方法，但不需要实现
* 不能包含任何变量

实现接口：实现接口所有方法

````go
package main

import "fmt"

type Usb interface {
	Start()
	Stop()
}

type Usb2 interface {
	Start()
	Stop()
}

type Phone struct {
	
}

func (p Phone) Start()  {
	fmt.Println("手机开始工作")
}

func (p Phone) Stop()  {
	fmt.Println("手机停止工作")
}

type Computer struct {
	
}

func (c Computer) Working (usb Usb2) {
	usb.Start()
	usb.Stop()
}

func main() {
	c := Computer{}
	phone :=Phone{}
	c.Working(phone)
}
````

### 使用细节

* 接口不能创建实例，但是可以指向一个实现了该接口的自定义类型的变量（实例）

  ````go
  package main

  import "fmt"

  type AInterface interface {
  	Say()
  }

  type Stu struct {
  	name string
  }

  func (stu Stu) Say() {
  	fmt.Println("stu Say()")
  }

  func main() {
  	//var a AInterface
  	//a.Say()  //无效的空指针

  	var stu Stu //结构体变量，实现你Say()  实现了 AInterface
  	var a AInterface = stu
  	a.Say()
  }
  ````

* 只要是自定义的数据类型，就可以实现接口，不仅仅是结构体类型

  ````go
  package main

  import "fmt"

  type AInterface interface {
  	Say()
  }

  type integer int

  func (i integer) Say() {
  	fmt.Println("integer say i", i)
  }

  func main() {
  	var i integer = 10
  	var b AInterface = i
  	b.Say()
  }
  ````

* 一个自定义类型可以实现多个接口

  ```go
  package main

  import "fmt"

  type AInterface interface {
  	Say()
  }

  type BInterface interface {
  	Hello()
  }

  type Monster struct {
  }

  func (m Monster) Hello()  {
  	fmt.Println("m Hello")
  }

  func (c Monster) Say()  {
  	fmt.Println("m Say")
  }

  func main() {
  	// 同时实现两个接口
  	var monster Monster
  	var a2 AInterface = monster
  	var b2 BInterface = monster
  	a2.Say()
  	b2.Hello()
  }
  ```

* 一个接口可以继承多个接口，但使用时均要实现

  ````go
  package main

  import "fmt"

  type AInterface interface {
  	test01()
  }

  type BInterface interface {
  	test02()
  }

  type MainInterface interface {
  	AInterface
  	BInterface
  	test03()
  }

  type Monster struct {
  }

  func (m Monster) test01() {
  	fmt.Println("test01")
  }

  func (m Monster) test02() {
  	fmt.Println("test02")
  }

  func (m Monster) test03() {
  	fmt.Println("test03")
  }

  func main() {
  	var stu Monster
  	var a AInterface = stu
  	a.test01()
  	var b BInterface = stu
  	b.test02()
  	var c MainInterface = stu
  	c.test01()
  	c.test02()
  	c.test03()
  }
  ````

## 最佳实践

* 对结构体切片的排序

  ````go
  package main

  import (
  	"fmt"
  	"math/rand"
  	"sort"
  )

  //1.声明一个Hero结构体
  type Hero struct {
  	Name string
  	Age  int
  }

  // 2. 声明一个Hero结构体对切片类型
  type HeroSlice []Hero

  // 3. 实现Interface接口
  func (hs HeroSlice) Len() int {
  	return len(hs)
  }
  func (hs HeroSlice) Less(i, j int) bool {
  	return hs[i].Age > hs[j].Age // 按年龄排序
  }
  func (hs HeroSlice) Swap(i, j int) {
  	tmp := hs[i]
  	hs[i] = hs[j]
  	hs[j] = tmp
  }

  func main() {
  	var intSlices = []int{1, 21, 21, 12, 143, 4, 32}
  	sort.Ints(intSlices)
  	fmt.Println(intSlices)

  	//对结构体切片排序
  	var heroes HeroSlice
  	for i := 0; i < 10; i++ {
  		hero := Hero{
  			Name: fmt.Sprintf("英雄编号%d", rand.Intn(999)),
  			Age:  rand.Intn(999),
  		}
  		heroes = append(heroes, hero)
  	}

  	// 排序前的顺序
  	fmt.Println("排序前的信息：")
  	for _, v := range heroes {
  		fmt.Println(v.Name, v.Age)
  	}
  	sort.Sort(heroes)

  	// 排序后的顺序
  	fmt.Println("排序后的信息：")
  	for _, v := range heroes {
  		fmt.Println(v.Name, v.Age)
  	}
  }
  ````


## 接口实现和集成的关系

1. 接口对继承的补充

````go
package main

import "fmt"

//猴子结构体
type Monkey struct {
	Name string
}

type BirdAble interface {
	Fly()
}

func (this *Monkey) climbing() {
	fmt.Println(this.Name, "生来会爬树")
}

//小猴子结构体
type LittleMonkey struct {
	Monkey
}

//小猴子实现飞翔
func (this *LittleMonkey) Fly() {
	fmt.Println(this.Name, "学会了飞翔")
}

func main() {
	// 创建小猴子
	monkey := LittleMonkey{
		Monkey{
			Name: "悟空",
		},
	}
	monkey.climbing()
	monkey.Fly()
}
````

### 接口和继承解决的问题不同

* 继承的价值在于解决代码的复用性和可维护性
* 接口的价值在于设计好各种规范，让其他自定义类型去实现这些代码方法
* 继承是is - a，接口是like - a 

## 多态

变量或实例具有多种形态

### 多态参数

````go
package main

import "fmt"

type Usb interface {
	connection()
}

type Phone struct {
	Name string
}

type PC struct {
	Name string
}

func (this Phone) connection()  {
	fmt.Println(this.Name,"链接了")
}

func (this PC) connection()  {
	fmt.Println(this.Name,"链接了")
}

func Working(obj Usb)  {
	obj.connection()
}

func main() {
	phone := Phone{Name: "华为手机"}
	Working(phone)
	pc := PC{Name: "mac电脑"}
	Working(pc)
}
````
