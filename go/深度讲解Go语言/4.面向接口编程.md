package rel

import (
	"net/http"
	"net/http/httputil"
	"time"
)

type Retriever struct {
	UserAgent string
	Timeout   time.Duration
}

func (r Retriever) Get(url string) string {
	resp, err := http.Get(url)
	if err != nil {
		panic(err)
	}

	result, err := httputil.DumpResponse(resp, true)
	if err != nil{
		panic(err)
	}
	defer resp.Body.Close()

	return string(result)
}

## 接口的类型

```go
package main

import (
	"fmt"
	"gotest/retriever/mock"
	"time"

	//"gotest/retriever/mock"
	"gotest/retriever/rel"
)

type Retriever interface {
	Get(url string) string
}

func download(r Retriever) string {
	return r.Get("https://www.baidu.com")
}

func inspect(r Retriever){
	fmt.Printf("%T %v\n", r, r)
	switch v := r.(type) {
	case mock.Retriever:
		fmt.Println("Contents:", v.Contents)
	case *rel.Retriever:
		fmt.Println("UserAgent:", v.UserAgent)
	}
}

func main() {
	var r Retriever
	r = mock.Retriever{"这是请求网页获得的内容"}
	fmt.Printf("%T %v\n", r, r)

	r = rel.Retriever{
		UserAgent: "Mozilla/5.0",
		Timeout: time.Minute,
	}
	fmt.Printf("%T %v\n", r, r)
	inspect(r)

	if mockRetriever, ok := r.(mock.Retriever); ok{//取得interface的类型
		fmt.Println(mockRetriever.Contents)
	} else {
		fmt.Println("not a mock retriever")
	}
}
```

* 接口变量自带指针
* 接口变量同样采用值传递，几乎不需要使用接口的指针
* 指针接收者，只能以指针方式传递，值接收者都可以

## 接口组合

```go
package main

import (
   "fmt"
   "gotest/retriever/mock"
   "time"

   //"gotest/retriever/mock"
   "gotest/retriever/rel"
)

type Retriever interface {
   Get(url string) string
}

type Poster interface {
   Post(url string, form map[string]string) string
}


func download(r Retriever) string {
   return r.Get("https://www.baidu.com")
}

func post(poster Poster){
   poster.Post("https://www.baidu.com", map[string]string{
      "name": "xin",
      "course": "golang",
   })
}

type RetrieverPoster interface {
   Retriever
   Poster
}


const url = "https://www.baidu.com"
func session(s RetrieverPoster) string{
   s.Post(url, map[string]string{
      "contents": "假的请求",
   })
   return s.Get(url)

}
```

````go
package mock

// 没有出现任何被实现接口的说明

type Retriever struct {
	Contents string
}

func (r *Retriever) Post(url string, form map[string]string) string { // 使用值传递无法修改对象的值，应该使用指针
	r.Contents = form["Contents"]
	return "ok"
}

func (r *Retriever) Get(url string) string{
	return r.Contents
}
````

## 自带接口

* `Stringer`

  ````go
  package main
  
  import "fmt"
  
  type Dog struct {
  	Name string
  	Age  int
  }
  
  func (d Dog) String() string {
  	return fmt.Sprintf("%s-%d", d.Name, d.Age)
  }
  
  func main() {
  	dog := Dog{"hahh", 12}
  	fmt.Println(dog)
  }
  ````

* 文件打印

  ```go
  package main
  
  import (
     "bufio"
     "fmt"
     "io"
     "os"
     "strings"
  )
  
  func printFile(filename string) {
     file, err := os.Open(filename)
     if err != nil {
        panic(err)
     }
     printFileContents(file)
  }
  
  func printFileContents(reader io.Reader) {
     scanner := bufio.NewScanner(reader)
  
     for scanner.Scan() {
        fmt.Println(scanner.Text())
     }
  }
  
  func main() {
     printFile("./go.mod")
     s := `dsdhasfdk
  
  
        j3q24ohfsfb`
  
     printFileContents(strings.NewReader(s))
  }
  ```
