# 内建容器

## 数组

### 多维数组

````go
package main

import "fmt"

func main() {
	var grid [4][5]int
	fmt.Println(grid)
}
````

* 数组是值类型，**值传递**

  ```go
  package main
  
  import "fmt"
  
  func printArray(arr [5]int){
  	for i,v := range(arr){
  		fmt.Println(i, v)
  		arr[0] = 100
  	}
  }
  
  
  func main() {
  	arr3 := [...]int{1, 2, 3, 4, 5}
  
  	printArray(arr3)
  	fmt.Println(arr3)
  }
  ```

* `[10]int`和`[11]int`是不同的类型

### 使用指针引用传递

```go
package main

import "fmt"

func printArray(arr *[5]int){
	for i,v := range(arr){
		fmt.Println(i, v)
		arr[0] = 100
	}
}


func main() {
	arr3 := [...]int{1, 2, 3, 4, 5}

	printArray(&arr3)
	fmt.Println(arr3)
}
```

## 切片

* 修改切片会修改源数组，本质是对底层数据的一个视图 

  ````go
  package main
  
  import "fmt"
  
  func updateSlice(s []int){
  	s[0] = 100
  }
  
  func main() {
  	arr := [...]int{1, 2, 3, 2, 4, 3, 234, 235, 423}
  	fmt.Println("arr[2:6]=", arr[2:6])
  	fmt.Println("arr[:6]=", arr[:6])
  	s1 := arr[2:]
  	fmt.Println("arr[2:]=", s1)
  	s2 := arr[:]
  	fmt.Println("arr[:]=", s2)
  	fmt.Println("------")
  	updateSlice(s1)
  	fmt.Println(s1)
  	fmt.Println(arr)
  	updateSlice(s2)
  	fmt.Println(s2)
  	fmt.Println(arr)
  	/*
  	[100 2 4 3 234 235 423]
  	[1 2 100 2 4 3 234 235 423]
  	[100 2 100 2 4 3 234 235 423]
  	[100 2 100 2 4 3 234 235 423]
  	*/
  }
  ````

* 取不到的会读取到源数据的


## Map

### 基础

```go
package main

import "fmt"

func main() {
	m := map[string]string{
		"name": "nnnnn",
		"site": "xin.com",
	}
	fmt.Println(m)

	m2 := make(map[string]int) // empty map
	fmt.Println(m2)

	var m3 map[string]int// nil
	fmt.Println(m3)

	//遍历
	for k, v := range m {
		fmt.Println(k, v)  //无序的
	}

	bing := m["bing"]
	fmt.Println(bing) //输出一个零值

	bing, ok := m["bing"]
	fmt.Println(bing, ok)

	if bing, ok := m["bing"]; ok{
		fmt.Println(bing)
	}else {
		fmt.Println("key does not exist")
	}

	delete(m, "name")
	fmt.Println(m)
}
```

* slice，map，function不能作为key 
* struct类型不饱饭上述字段也能作为key

## 字符与字符串

```go
package main

import (
	"fmt"
	"unicode/utf8"
)

func main() {
	s := "ok哈哈哈哈哈"
	fmt.Println(s)
	fmt.Println("%s\n", []byte(s) ) // [111 107 229 147 136 229 147 136 229 147 136 229 147 136 229 147 136]

	for _, b := range []byte(s){
		fmt.Printf("%X ", b) // 6F 6B E5 93 88 E5 93 88 E5 93 88 E5 93 88 E5 93 88
	}
	fmt.Println()

	for i, ch := range s{ // ch is a rune
		fmt.Printf("(%d %X)", i, ch) // (0 6F)(1 6B)(2 54C8)(5 54C8)(8 54C8)(11 54C8)(14 54C8)
	}
	fmt.Println()

	fmt.Println(utf8.RuneCountInString(s)) //7

	bytes := []byte(s)
	for len(bytes) > 0{
		ch, size := utf8.DecodeRune(bytes)
		bytes = bytes[size:]
		fmt.Printf("%c ", ch) // o k 哈 哈 哈 哈 哈
	}
	fmt.Println()

	for i, ch := range []rune(s) {
		fmt.Printf("(%d %c)", i, ch) // (0 o)(1 k)(2 哈)(3 哈)(4 哈)(5 哈)(6 哈)
	}
	fmt.Println()
}
```
