# go的web框架-gin

## gin的helloworld体验

````go
package main

import (
	"github.com/gin-gonic/gin"
	"net/http"
)

func main() {
	r := gin.Default()
	r.GET("/ping", func(c *gin.Context) {
		c.JSON(http.StatusOK, gin.H{
			"message": "pong",
		})
	})
	r.Run() // listen and serve on 0.0.0.0:8080
}
````

## 获取url中的变量

````go
package main

import (
	"github.com/gin-gonic/gin"
	"net/http"
)

func readEndpoint(c *gin.Context) {
	c.JSON(http.StatusOK, gin.H{
		"id": c.Param("id"),
		"action": c.Param("action"),
	})
}

func main() {
	router := gin.Default()

	v1 := router.Group("/v1")
	{
		v1.GET("/read/:id/:action", readEndpoint)
	}

	router.Run(":8088")
}
````

### 值的约束

````go
package main

import (
	"github.com/gin-gonic/gin"
	"net/http"
)

type Person struct {
	ID int `uri:"id" binding:"required"`
	Name string `uri:"name" binding:"required"`
}

func main() {
	router := gin.Default()
	router.GET("/:name/:id", func(c *gin.Context){
		var person Person
		if err := c.ShouldBindUri(&person); err != nil {
			c.Status(404)
			return
		}
		c.JSON(http.StatusOK, gin.H{
			"name": person.Name,
			"id": person.ID,
		})
	})

	router.Run(":8088")
}
````

## 获取get和post表单信息

````go
package main

import (
	"github.com/gin-gonic/gin"
	"net/http"
)

func main() {
	router := gin.Default()
	router.GET("/test", test)
	router.POST("/post_test", post_test)

	router.Run(":8088")
}

func post_test(c *gin.Context) {
	message := c.PostForm("message")
	nick := c.DefaultPostForm("nick", "家在河边喜欢浪")
	c.JSON(http.StatusOK, gin.H{
		"message": message,
		"nick":    nick,
	})
}

func test(c *gin.Context) {
	firstName := c.DefaultQuery("first", "xin") //取得并设置默认值
	lastName := c.Query("last")
	c.JSON(http.StatusOK, gin.H{
		"firstName": firstName,
		"lastName":  lastName,
	})
}
````

## 登录的表单验证

[位置](https://github.com/go-playground/validator)

`````go
package main

import (
	"fmt"
	"github.com/gin-gonic/gin"
	"net/http"
)

type LoginForm struct {
	User string `form:"user" json:"user" xml:"user" binding:"required,min=3,max=120"` //注意一个""中间是没有空格的
	Password string `form:"password" json:"password" xml:"password" binding:"required"`
}

func main() {
	router := gin.Default()

	router.POST("/login", login)

	router.Run(":8088")
}

func login(c *gin.Context) {
	var loginForm LoginForm
	if err := c.ShouldBind(&loginForm); err != nil{
		fmt.Println(err.Error())
		c.JSON(http.StatusBadRequest, gin.H{
			"error": err.Error(),
		})
		return
	}
	c.JSON(http.StatusOK, gin.H{
		"msg": "登录成功",
	})
}
`````
