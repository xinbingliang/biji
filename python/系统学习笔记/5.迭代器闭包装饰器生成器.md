## 闭包

```python
def test(number):
    def test_in(number_in):
        print(number_in + number)

    return test_in

T = test(10)
T(100)
```

### 闭包应用

```python
def line_conf(a, b):
    def line(x):
        return a*x + b

    return line

line1 = line_conf(1, 1)
line2 = line_conf(4, 5)

print(line1(5))
print(line2(5))
```

## 装饰器

对函数的装扮，装饰器在调用之前，已经完成了装饰

```python
def w1(func):
    def inner():
        print('---正在权限验证---')
        func()
    return inner

@w1
def f1():
    print('f1')

@w1
def f2():
    print('f2')


@w1
def f3():
    print('f3')


f1()
f2()
f3()
```

### 装饰有参数函数

```python
def fucn(funcName):
    def inner(a, b):
        funcName(a, b)
    
    return inner

@fucn
def test(a, b):
    print(a, b)

test(1, 2)
```

```python
def fucn(funcName):
    def inner(*args, **kwargs):
        funcName(*args, **kwargs)
    
    return inner

@fucn
def test(a, b):
    print(a, b)

test(1, 2)
```

### 带返回值的装饰器

```python
def fucn(funcName):
    def inner(*args, **kwargs):
        return funcName(*args, **kwargs)
    
    return inner

@fucn
def test(a, b):
    return 'aa'

print(test(1, 2))
```

### 带参数装饰器

```python
def time_func(pre='123'):
    def func(funcname):
        def func_in():
            print(pre)
            funcname()

        return func_in
    return func

@time_func('456')
def test():
    print("---test---")

test()
```

## 动态添加属性方法

###对象方法

```python
import types

class Person(object):
    def __init__(self, newName, newAge):
        self.name = newName
        self.age = newAge

p1 = Person('辛丙亮', 24)

def run(self, arg):
    print(self.name)
    print(arg)

p1.run = types.MethodType(run, p1)

p1.run('哈哈') #将self指向p1
```

### 添加静态方法

```python
import types

class Person(object):
    def __init__(self, newName, newAge):
        self.name = newName
        self.age = newAge

p1 = Person('辛丙亮', 24)

@staticmethod
def run():
    print('静态方法')

Person.run = run

Person.run()
```

###控制属性的定义

`__slots__()` 划定属性范围

```python
class Person(object):
    __slots__ = ('name', 'age')

    def __init__(self, newName, newAge):
        self.name = newName
        self.age = newAge

p1 = Person('辛丙亮', 24)

p1.name = '123'
p1.addr = '2344'
```

###生成斐波拉契数列

```python
def fib():
    print('--- start ---')
    a, b = 0, 1

    for i in range(100):
        yield b
        a, b = b, a+b

    print('--- end ---')

a = fib()
next(a)
```
