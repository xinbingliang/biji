## 用蓝图注册视图函数 

向应用中注册蓝图

````python
# app/__init__.py
from flask import Flask

def create_app():
    app = Flask(__name__)
    app.config.from_object('config')
    register_blueprint(app)
    return app

def register_blueprint(app):
    from app.web.book import web
    app.register_blueprint(web) # 用于注册蓝图对象
````

蓝图使用

````python
# app/web/book.py
from flask import Blueprint

web = Blueprint('web', __name__) # 模块名，所在的包或模块

@web.route('/book/search/<q>/<page>')
def search(q, page):
    """
    q: 查询的关键字
    page:页数
    :return:
    """
    isbn_or_key = is_isbn_or_key(q)

    if isbn_or_key == 'isbn':
        result = XinBook.search_by_isbn(q)
    else:
        result = XinBook.search_by_keyword(q)

    return jsonify(result)
    # return json.dumps(result), 200, {"content-type": "application/json"}
````

```python
# /app/web/book.py
from . import web
from flask import request
from app.forms.book import SearchForm

@web.route("/book/search")
def search():
    # 查询参数 POST参数 remote ip
    q = request.args['q']
    page = request.args['page'] # 可以取到form中设置的默认值

    # 验证层
    form = SearchForm(request.args)
    if form.validate():
        q = form.q.data
        page = form.page.data
    else:
        pass

    # a = request.args.to_dict() # 将不可变字典转换为可变字典
    return "search"
```
