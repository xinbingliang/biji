# 视图函数的路由

## 路由规则设置说明

### 显示路由配置

````python
if __name__ == '__main__':
    print(app.url_map)
    app.run(host='0.0.0.0', port='5000', debug=True)
````

### 调整访问方式

````python
from flask import Flask

app = Flask(__name__)

@app.route("/")
def index():
    """
    定义的视图函数
    :return:
    """
    return "Hello"

@app.route('/post_only', methods=['POST']) # 指定访问方式
def post_only():
    """
    只能post访问
    :return:
    """

if __name__ == '__main__':
    print(app.url_map)
    app.run(host='0.0.0.0', port='5000', debug=True)
````

###  同一路由或同一函数


## 路由提取参数与自定义路由转换器

### 参数提取

![](./images/1.png)

````python
from flask import Flask

app = Flask(__name__)

# @app.route("/index/<int:index_id>") # 转换器
@app.route("/index/<index_id>") # 转换器,默认使用字符串
def index(index_id):
    return str(index_id)

if __name__ == '__main__':
    print(app.url_map)
    app.run(host='0.0.0.0', port='5000', debug=True)
````

### 自定义转换器

````python
from flask import Flask
from werkzeug.routing import BaseConverter

app = Flask(__name__)

# 1. 定义自己的转换器
class RegexConverter(BaseConverter):
    def __init__(self, url_map, regex):
        # 调用父类的初始化方法
        super(RegexConverter, self).__init__(url_map)
        # 将正则表达式的参数保存到对象属性中, flask会使用这个属性进行路由的正则匹配
        self.regex = regex


# 2. 将自定义转换器添加到flask应用中
app.url_map.converters['re'] = RegexConverter

# 3. 使用
@app.route("/send/<re(r'1[34578]\d{9}'):mobile>")
def send_sms(mobile):
    return mobile

if __name__ == '__main__':
    print(app.url_map)
    app.run(host='0.0.0.0', port='5000', debug=True)
````

### 进阶使用

````python
from flask import Flask, redirect, url_for
from werkzeug.routing import BaseConverter

app = Flask(__name__)

class MobileConverter(BaseConverter):
    def __init__(self, url_map):
        super(MobileConverter, self).__init__(url_map)
        self.regex = r'1[34578]\d{9}'

    def to_python(self, value):
        """
        正则匹配后进行匹配内容转换
        :param value:
        :return:
        """
        return value

    def to_url(self, value):
        """
        路由跳转时会执行
        :param value:
        :return:
        """
        return '15102724458'

app.url_map.converters['mobile'] = MobileConverter

@app.route("/send/<mobile:mobile>")
def send_sms(mobile):
    return mobile

@app.route("/index")
def index():
    url = url_for('send_sms', mobile="15102724511")
    return redirect(url)

if __name__ == '__main__':
    print(app.url_map)
    app.run(host='0.0.0.0', port='5000', debug=True)
````
